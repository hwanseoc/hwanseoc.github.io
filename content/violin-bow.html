Model (plaintext)
N + T = W + F1 + F4

d_N * N - d_W * W - d_F1 * F1 + d_F4 * F4 = 0
Algorithm
R = d_W * W - d_N * N
If R > 0:  F1 = 0,  F4 = R / d_F4
If R < 0:  F4 = 0,  F1 = -R / d_F1
If R = 0:  F1 = 0,  F4 = 0
T = W + F1 + F4 - N
